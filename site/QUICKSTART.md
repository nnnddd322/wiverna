```typescript
// src/services/lectureService.ts
import { supabase } from '../lib/supabase';

export const lectureService = {
  async getLecturesByDiscipline(disciplineId: string) {
    const { data, error } = await supabase
      .from('lectures')
      .select('*')
      .eq('discipline_id', disciplineId)
      .eq('status', 'published')
      .order('order_index');
    
    if (error) throw error;
    return data;
  },

  async createLecture(disciplineId: string, title: string, content: any) {
    const { data, error } = await supabase
      .from('lectures')
      .insert({
        discipline_id: disciplineId,
        title,
        content,
        type: 'article',
        status: 'draft'
      })
      .select()
      .single();
    
    if (error) throw error;
    return data;
  },

  async updateLecture(id: string, updates: any) {
    const { data, error } = await supabase
      .from('lectures')
      .update(updates)
      .eq('id', id)
      .select()
      .single();
    
    if (error) throw error;
    return data;
  },

  async publishLecture(id: string) {
    return this.updateLecture(id, { status: 'published' });
  }
};
```
